代码示例

// 基本使用
import { success, info, warning, error } from 'console-charm';

success('操作成功');
info('提示信息');
warning('警告信息');
error('错误信息');

// 构建信息
import { buildInfo } from 'console-charm';

buildInfo({
    version: '1.0.0',
    environment: 'production',
    buildDate: new Date(),
    branch: 'main',
    commit: 'abc123def456'
});
